344/*
345 * Auxiliary function to remove an outgoing edge (with a given destination (d))
346 * from a vertex (this).
347 * Returns true if successful, and false if such edge does not exist.
348 */
349template <class T>
350bool Vertex<T>::removeEdge(T in) {
351 bool removedEdge = false;
352 auto it = adj.begin();
353 while (it != adj.end()) {
354 Edge<T> *edge = *it;
355 Vertex<T> *dest = edge->getDest();
356 if (dest->getID() == in) {
357 it = adj.erase(it);
358 deleteEdge(edge);
359 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
360 }
361 else {
362 it++;
363 }
364 }
365 return removedEdge;
366}

369 * Auxiliary function to remove an outgoing edge of a vertex.

371template <class T>
372void Vertex<T>::removeOutgoingEdges() {
373 auto it = adj.begin();
374 while (it != adj.end()) {
375 Edge<T> *edge = *it;
376 it = adj.erase(it);
377 deleteEdge(edge);
378 }
379}

476template <class T>
477void Vertex<T>::deleteEdge(Edge<T> *edge) {
478 Vertex<T> *dest = edge->getDest();
479 // Remove the corresponding edge from the incoming list
480 auto it = dest->incoming.begin();
481 while (it != dest->incoming.end()) {
482 if ((*it)->getOrig()->getID() == id) {
483 it = dest->incoming.erase(it);
484 }
485 else {
486 it++;
487 }
488 }
489 delete edge;
490}

606/*
607 * Removes a vertex with a given content (in) from a graph (this), and
608 * all outgoing and incoming edges.
609 * Returns true if successful, and false if such vertex does not exist.
610 */
611template <class T>
612bool Graph<T>::removeVertex(const T &in) {
613 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
614 if ((*it)->getID() == in) {
615 auto v = *it;
616 v->removeOutgoingEdges();
617 for (auto u : vertexSet) {
618 u->removeEdge(v->getID());
619 }
620 vertexSet.erase(it);
621 delete v;
622 return true;
623 }
624 }
625 return false;
626}

628/*
629 * Removes an edge from a graph (this).
630 * The edge is identified by the source (sourc) and destination (dest) contents.
631 * Returns true if successful, and false if such edge does not exist.
632 */
633template <class T>
634bool Graph<T>::addBidirectionalEdge(const std::string &source, const std::string &dest, double distance, std::string label) {
635 auto v1 = findVertex(source);
636 auto v2 = findVertex(dest);
637 if (v1 == nullptr || v2 == nullptr) return false;
638 auto e1 = v1->addEdge(v2, distance, label);
639 auto e2 = v2->addEdge(v1, distance, label);
640 e1->setReverse(e2);
641 e2->setReverse(e1);
642 return true;
643}
