175/*
176 * Auxiliary function to remove an outgoing edge (with a given destination (d))
177 * from a vertex (this).
178 * Returns true if successful, and false if such edge does not exist.
179 */
180template <class T>
181bool Vertex<T>::removeEdge(T in) {
182 bool removedEdge = false;
183 auto it = adj.begin();
184 while (it != adj.end()) {
185 Edge<T> *edge = *it;
186 Vertex<T> *dest = edge->getDest();
187 if (dest->getID() == in) {
188 it = adj.erase(it);
189 deleteEdge(edge);
190 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
191 }
192 else {
193 it++;
194 }
195 }
196 return removedEdge;
197}

199/*
200 * Auxiliary function to remove an outgoing edge of a vertex.
201 */
202template <class T>
203void Vertex<T>::removeOutgoingEdges() {
204 auto it = adj.begin();
205 while (it != adj.end()) {
206 Edge<T> *edge = *it;
207 it = adj.erase(it);
208 deleteEdge(edge);
209 }
210}

307template <class T>
308void Vertex<T>::deleteEdge(Edge<T> *edge) {
309 Vertex<T> *dest = edge->getDest();
310 // Remove the corresponding edge from the incoming list
311 auto it = dest->incoming.begin();
312 while (it != dest->incoming.end()) {
313 if ((*it)->getOrig()->getID() == id) {
314 it = dest->incoming.erase(it);
315 }
316 else {
317 it++;
318 }
319 }
320 delete edge;
321}

437/*
438 * Removes a vertex with a given content (in) from a graph (this), and
439 * all outgoing and incoming edges.
440 * Returns true if successful, and false if such vertex does not exist.
441 */
442template <class T>
443bool Graph<T>::removeVertex(const T &in) {
444 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
445 if ((*it)->getID() == in) {
446 auto v = *it;
447 v->removeOutgoingEdges();
448 for (auto u : vertexSet) {
449 u->removeEdge(v->getID());
450 }
451 vertexSet.erase(it);
452 delete v;
453 return true;
454 }
455 }
456 return false;
457}

459/*
460 * Removes an edge from a graph (this).
461 * The edge is identified by the source (sourc) and destination (dest) contents.
462 * Returns true if successful, and false if such edge does not exist.
463 */
464template <class T>
465bool Graph<T>::addBidirectionalEdge(const std::string &source, const std::string &dest, double distance, std::string label) {
466 auto v1 = findVertex(source);
467 auto v2 = findVertex(dest);
468 if (v1 == nullptr || v2 == nullptr) return false;
469 auto e1 = v1->addEdge(v2, distance, label);
470 auto e2 = v2->addEdge(v1, distance, label);
471 e1->setReverse(e2);
472 e2->setReverse(e1);
473 return true;
474}
