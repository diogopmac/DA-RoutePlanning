91void MutablePriorityQueue<T>::heapifyUp(unsigned i) {
92 auto x = H[i];
93 while (i > 1 && *x < *H[parent(i)]) {
94 set(i, H[parent(i)]);
95 i = parent(i);
96 }
97 set(i, x);

104template <class T>
105void MutablePriorityQueue<T>::heapifyDown(unsigned i) {
106 auto x = H[i];
107 while (true) {
108 unsigned k = leftChild(i);
109 if (k >= H.size())
110 break;
111 if (k+1 < H.size() && *H[k+1] < *H[k])
112 ++k; // right child of i
113 if ( ! (*H[k] < *x) )
114 break;
115 set(i, H[k]);
116 i = k;
117 }
118 set(i, x);
119}
